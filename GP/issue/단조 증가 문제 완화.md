
기회가 없었던 판을 이상한 비율로 판단하지 못하게 막기

```
private static int CR(float v) => Mathf.Clamp(Mathf.RoundToInt(v), -10, 10);

private static PlayerProfile DeltaProcess(BehaviorSummary s)
{
    float farm = s.farmingPerOpp - 0.5f;     // [-0.5, +0.5]
    float talk = s.talkPerOpp - 0.5f;
    float kill = s.killPerOpp - 0.5f;
    float avoid = s.avoidPerOpp - 0.5f;

    // radiation은 opportunity가 있을 때만 의미가 있으니 중심값은 0.5 기준이 아니라 0 기준으로 써도 됨
    float rad = (s.radiationOpportunityCount > 0) ? (s.radiationExposurePerOpp - 0.5f) : 0f;

    float detour = s.detourRate - 0.25f; // 우회 baseline

    int dGreedy     = CR(farm * 12f);
    int dSocial     = CR(talk * 12f);

    int dCurious    = CR(detour * 12f + farm * 4f + talk * 4f);
    int dDiscipline = CR((-detour) * 12f);

    if (s.endType == ExpedEndType.Abort) dDiscipline = Mathf.Clamp(dDiscipline - 5, -10, 10);

    int dCruel      = CR(kill * 14f);
    int dMercy      = CR((avoid - kill) * 10f + talk * 6f);

    int dRisk       = CR((kill - avoid) * 8f + rad * 10f);
    int dCaution    = CR((avoid - kill) * 8f - rad * 8f);

    return new PlayerProfile
    {
        mercy = dMercy,
        greedy = dGreedy,
        curious = dCurious,
        discipline = dDiscipline,
        risk = dRisk,
        social = dSocial,
        cruel = dCruel,
        caution = dCaution
    };
}
```
